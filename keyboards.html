from aiogram.types import InlineKeyboardMarkup, InlineKeyboardButton, WebAppInfo
from config import PAY_LINK_ACADEMY, PAY_LINK_PRIVATE

import buttons_text as btn

def start_markup() -> InlineKeyboardMarkup:
    """
    Стартовая клавиатура
    :return: InlineKeyboardMarkup
    """
    buttons = [
        [InlineKeyboardButton(text=btn.buy_access_button, callback_data='buy_access')],
        [InlineKeyboardButton(text=btn.personal_area_button, callback_data='personal_area')],
        [InlineKeyboardButton(text=btn.support_button, url='google.com')]
    ]

    kb = InlineKeyboardMarkup(inline_keyboard=buttons)

    return kb


def already_have_access() -> InlineKeyboardMarkup:
    """
    Отправляет на стартовое сообщение
    :return: InlineKeyboardMarkup
    """
    markup = [
        [InlineKeyboardButton(text=btn.back_button, callback_data='start_message')]
    ]

    kb = InlineKeyboardMarkup(inline_keyboard=markup)

    return kb


# client/academy
def academy_pay_kb(user_id) -> InlineKeyboardMarkup:
    """
    Клавиатура с оплатой курса Academy
    :param user_id: user_id пользователя
    :return: InlineKeyboardMarkup
    """
    pay_link: str = PAY_LINK_ACADEMY + f'{user_id}'
    markup = [
        [InlineKeyboardButton(text=btn.pay_button, web_app=WebAppInfo(url=pay_link))],
        [InlineKeyboardButton(text=btn.support_button, url='google.com')],
        [InlineKeyboardButton(text=btn.back_button, callback_data='start_message')]
    ]

    kb = InlineKeyboardMarkup(inline_keyboard=markup)

    return kb
